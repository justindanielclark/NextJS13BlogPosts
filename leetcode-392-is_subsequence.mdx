function isSubsequence(s: string, t: string): boolean {
  let curr = 0;
  for (let i = 0; i < t.length; i++) {
    if (t.substring(i, i + 1) === s.substring(curr, curr + 1)) curr++;
  }
  return curr >= s.length;
}
